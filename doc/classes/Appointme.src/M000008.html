  def self.get_range_of_time_to_show time
    time = Chronic.parse( time, :context => :past) if time.kind_of? String
    sec, min, hour  = time.to_a[0,3]

    hour = 23 if hour == 0
    min = 59 if min == 0
    sec = 59 if sec == 0

    time..Time.mktime(time.year, time.month, time.day, hour, min, sec)
  end
