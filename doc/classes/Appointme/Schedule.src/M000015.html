    def show(what)
      show_appointments = Appointme.get_range_of_time_to_show(what)
      select do |appointment|
        show_appointments.include? appointment.when 
        end.sort_by do |appointment|
          appointment.when
          end
    rescue
      puts "errrr: #{$!}", $@
    end
